import java.util.Scanner;
import javax.swing.JOptionPane;
public class Tarea17{
	public static float conversionMxnaUsd(float mxn){
		float usd;
		System.out.println("\nHas elegido la conversion de MXN a USD");
		System.out.println("\n*** El valor de 1 MXN es de 0.05 USD ***");
		usd = mxn * 0.05f;
		System.out.println();
		System.out.println("\nEl valor de " + mxn + " MXN es de " + usd + " USD");
		return 0;
	}
	public static double conversionMxnaEuro(float mxn){
		float euro;
		System.out.println("\nHas elegido la conversion de MXN a Euro");
		System.out.println("\n*** El valor de 1 MXN es de 0.05 Euro ***");
		euro = mxn * 0.043f;
		System.out.println();
		System.out.println("\nEl valor de " + mxn + " MXN es de " + euro + " Euro");
		return 0;
	}
	public static double conversionMxnaYen(float mxn){
		float yen;
		System.out.println("\nHas elegido la conversion de MXN a Yen");
		System.out.println("\n*** El valor de 1 MXN es de 5.52 Yen ***");
		yen = mxn * 5.52f;
		System.out.println();
		System.out.println("\nEl valor de " + mxn + " MXN es de " + yen + " Yen");
		return 0;
	}
	public static double conversionMxnaHk(float mxn){
		float hk;
		System.out.println("\nHas elegido la conversion de MXN a HK");
		System.out.println("\n*** El valor de 1 MXN es de 0.39 HK ***");
		hk = mxn * 0.39f;
		System.out.println();
		System.out.println("\nEl valor de " + mxn + " MXN es de " + hk + " HK");
		return 0;
	}
	public static void main(String[] args){
		Scanner lec = new Scanner(System.in);
		int salida = 0;
		float mxn;
		do{
			String[] options = {"Dolar", "Euro", "Yen", "Dolar de Honk Kong"};
			System.out.println();		//por estetica
			System.out.print("Ingrese el monto de MXN que desee convertir: ");
			mxn = lec.nextFloat();
			System.out.println();		//por estetica
			int opcion = JOptionPane.showOptionDialog(null, "Que conversion desea aplicar?",
				"Pulse una opcion",
				JOptionPane.DEFAULT_OPTION, JOptionPane.INFORMATION_MESSAGE, null, options, options[0]);
			switch (opcion){
				case 0:
					conversionMxnaUsd(mxn);
				break;
				case 1:
					conversionMxnaEuro(mxn);
				break;
				case 2:
					conversionMxnaYen(mxn);
				break;
				case 3:
					conversionMxnaHk(mxn);
				break;
			}
			System.out.print("\n\nDeseas volver a ejecutar el programa? (| 1 = Si || 0 = No |): ");
			salida = lec.nextInt();
		}while(salida != 0);
	}
}
