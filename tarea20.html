import java.util.Scanner;
public class Tarea20{
	public static void main(String[] args){
		Scanner lec = new Scanner(System.in);
		int repeticiones = 0;
		System.out.println("\nLa forma de escribirlo es matriz[N][M]\n");
		System.out.print("Ingresa N: ");
		int n = lec.nextInt();
		System.out.print("Ingresa M: ");
		int m = lec.nextInt();
		int[][] array = new int[n][m];
		System.out.println("\nPor favor ingrese los numeros para llenar el vector");
		System.out.println();		//por estetica
		for(int i = 0 ; i < n ; i++){		//crear la matriz
			for(int j = 0 ; j < m ; j++){
				System.out.print("Ingresa el numero entero en [" + i + "][" + j + "]: ");
				array[i][j] = lec.nextInt();
			}
		}
		System.out.print("\nPor favor ingrese el numero del que desea saber sus repeticiones: ");
		int valorABuscar = lec.nextInt();
		for(int i = 0 ; i < n ; i++){		//crear la matriz
			for(int j = 0 ; j < m ; j++){
				if(array[i][j] == valorABuscar){
					repeticiones++;
				}
			}
		}
		System.out.println("\nEl numero " + valorABuscar + " se ha repetido: " + repeticiones + " veces!");
		System.out.println();		//por estetica
		System.out.println("A continuacion se imprimira la matriz\n");
		for(int i = 0 ; i < n ; i++){		//imprimir la matriz
			for(int j = 0 ; j < m ; j++){
				System.out.println("El valor de [" + i + "][" + j + "] es: " + array[i][j]);
			}
		}
	}
}
