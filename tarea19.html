import java.util.Scanner;
public class Tarea19{
	public static int[] array = new int[10];
	public static int repeticiones = 0;
	public static void crearArrayInt(){
		Scanner lec = new Scanner(System.in);
		System.out.println("\nPor favor ingrese 10 numeros para llenar el vector");
		System.out.println();		//por estetica
		for(int x = 0 ; x < array.length ; x++){
			System.out.print("Ingresa el numero entero " + (x + 1) + ": ");
			array[x] = lec.nextInt();
		}
	}
	public static void mostrarArray(){
		System.out.println();		//por estetica
		for(int x = 0 ; x < array.length ; x++){
			System.out.println("El numero entero " + (x + 1) + " es: " + array[x]);
		}
	}
	public static void buscarEnArray(){
		Scanner lec = new Scanner(System.in);
		System.out.print("\nIngrese el numero que desea saber cuantas repeticiones tiene dentro del vector: ");
		int valorABuscar = lec.nextInt();
		for(int x = 0 ; x < array.length ; x++){
			if(array[x] == valorABuscar){
				repeticiones++;
			}
		}
		System.out.println("\nEl numero " + valorABuscar + " se ha repetido: " + repeticiones + " veces!");
	}
	public static void main(String[] args){
		Scanner lec = new Scanner(System.in);
		crearArrayInt();
		buscarEnArray();
		System.out.print("\nDesea mostrar el vector? (| Si = 1 || No = 0 |): ");
		int mostrar = lec.nextInt();
		if(mostrar == 1){
			mostrarArray();
		}
		else{
			System.out.println("\nGracias por usar el programa!");
		}
	}
}
