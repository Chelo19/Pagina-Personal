import java.util.Scanner;
import javax.swing.JOptionPane;
public class Tarea18{
	public static long exponenciar(long n, long exponente){
		long r = 1;
		for(int x = 0 ; x < exponente ; x++){
			r = r * n;
		}
		return r;
	}
	public static void main(String[] args){
		Scanner lec = new Scanner(System.in);
		long opcion, resultado, exponente;
		System.out.print("\nIngrese el numero de datos que desee ver de la serie | 1, 8, 243, 16384, 1953125 |: ");
		opcion = lec.nextInt();
		for(int n = 1 ; n <= opcion ; n++){
			exponente = (2 * n) - 1;
			resultado = exponenciar(n, exponente);
			System.out.println("\nN: " + n);
			System.out.println("Resultado: " + resultado);
		}
	}
}
