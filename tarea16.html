import java.util.Scanner;
import javax.swing.JOptionPane;
public class Tarea16{
	public static float conversionCaF(){
		Scanner lec = new Scanner(System.in);
		float gradosC, gradosF;
		System.out.println();		//por estetica
		System.out.print("Por favor, ingrese la cantidad de grados en C: ");
		gradosC = lec.nextFloat();
		System.out.println();		//por estetica
		gradosF = (gradosC * 1.8f) + 32f;
		System.out.println("El resultado es de: " + gradosF + " grados Farenheit");
		System.out.println();		//por estetica
		return 0;
	}
	public static float conversionFaC(){
		Scanner lec = new Scanner(System.in);
		float gradosC, gradosF;
		System.out.println();		//por estetica
		System.out.print("Por favor, ingrese la cantidad de grados en F: ");
		gradosF = lec.nextFloat();
		gradosC = (gradosF - 32f) * 0.55f;
		System.out.println();		//por estetica
		System.out.println("El resultado es de: " + gradosC + " grados Celsius");
		System.out.println();		//por estetica
		return 0;
	}
	public static void main(String[] args){
		Scanner lec = new Scanner(System.in);
		int salida = 0;
		do{
			String[] options = {"C -> F", "F -> C"};
			int x = JOptionPane.showOptionDialog(null, "Que conversion desea aplicar?",
				"Pulse una opcion",
				JOptionPane.DEFAULT_OPTION, JOptionPane.INFORMATION_MESSAGE, null, options, options[0]);
			if(x == 0){
				System.out.println();		//por estetica
				System.out.println("Has decidido aplicar la conversion de C a F");
				conversionCaF();
			}
			else{
				System.out.println();		//por estetica
				System.out.println("Has decidido aplicar la conversion de F a C");
				conversionFaC();
			}
			System.out.println();		//por estetica
			System.out.print("Deseas volver a ejecutar el programa? (| 1 = Si || 0 = No |): ");
			salida = lec.nextInt();
		}while(salida != 0);
	}
}
